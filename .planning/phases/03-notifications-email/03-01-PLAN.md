```typescript
import { Resend } from 'resend'
import { renderToStaticMarkup } from 'react-dom/server'
import { WorkOrderAssignedEmail } from '@/emails/work-order-assigned'
import { MaintenanceReminderEmail } from '@/emails/maintenance-reminder'
import { LowStockAlertEmail } from '@/emails/low-stock-alert'

const resend = new Resend(process.env.RESEND_API_KEY)

const FROM = process.env.RESEND_FROM_EMAIL ?? 'Korvia <notifications@korvia.app>'

export async function sendWorkOrderAssignedEmail(params: {
  to: string
  recipientName: string
  workOrderNumber: number
  workOrderTitle: string
  workOrderPriority: string
  dueDate?: Date | null
  organizationName: string
}): Promise<void> {
  const html = renderToStaticMarkup(WorkOrderAssignedEmail(params))
  await resend.emails.send({
    from: FROM,
    to: [params.to],
    subject: `[Korvia] Bon de travail #${params.workOrderNumber} — ${params.workOrderTitle}`,
    html,
  })
}

export async function sendMaintenanceReminderEmail(params: {
  to: string[]
  planName: string
  assetName?: string | null
  nextDueAt: Date
  organizationName: string
}): Promise<void> {
  const html = renderToStaticMarkup(MaintenanceReminderEmail(params))
  await resend.emails.send({
    from: FROM,
    to: params.to,
    subject: `[Korvia] Rappel maintenance — ${params.planName} dans 48h`,
    html,
  })
}

export async function sendLowStockAlertEmail(params: {
  to: string[]
  partName: string
  partNumber?: string | null
  quantityOnHand: number
  quantityMin: number
  organizationName: string
}): Promise<void> {
  const html = renderToStaticMarkup(LowStockAlertEmail(params))
  await resend.emails.send({
    from: FROM,
    to: params.to,
    subject: `[Korvia] Alerte stock faible — ${params.partName}`,
    html,
  })
}
```

Notes:
- Use `renderToStaticMarkup` from `react-dom/server` (NOT React Email's render — it is not installed)
- The Resend singleton is module-level (lazy init on first call — same pattern as `src/lib/db.ts`)
- `RESEND_FROM_EMAIL` env var with fallback so dev environments without the var don't crash
- All three functions are `async` and return `Promise<void>` — callers can fire-and-forget with `.catch(console.error)` to avoid blocking the Server Action

| Boundary | Description |
|----------|-------------|
| Server Action → Resend API | Outbound HTTP call with secret API key |
| env vars → runtime | RESEND_API_KEY and RESEND_FROM_EMAIL sourced from environment |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01-01 | Information Disclosure | RESEND_API_KEY in env | mitigate | Never commit real key; .env.example contains placeholder value only; .gitignore covers .env |
| T-03-01-02 | Spoofing | FROM address in emails | mitigate | FROM is set from RESEND_FROM_EMAIL env var (server-side only); callers cannot override it |
| T-03-01-03 | Denial of Service | Resend API rate limits | accept | Send functions are fire-and-forget on transactional events (1 email per action); no bulk sending in this plan |
| T-03-01-04 | Information Disclosure | Email template content | accept | Templates contain only data passed by caller; no DB queries in templates; caller is authenticated Server Action |
</threat_model>
