```typescript
import type { SubscriptionPlan } from '@/generated/prisma/enums'
// (add to existing imports block at top of file — check if SubscriptionPlan is already imported)

const ACTIVE_STATUSES = ['active', 'trialing'] as const

export async function requirePlan(plans: SubscriptionPlan[]) {
  const membership = await getOrganizationMembership()
  if (!membership) throw new Error('Unauthorized')

  const sub = membership.organization.subscription
  // null subscription OR non-active status → effective plan is 'starter'
  const effectivePlan: SubscriptionPlan =
    sub && ACTIVE_STATUSES.includes(sub.status as (typeof ACTIVE_STATUSES)[number])
      ? sub.plan
      : 'starter'

  return {
    membership,
    subscription: sub,
    effectivePlan,
    hasAccess: plans.includes(effectivePlan),
  }
}
```

| `requirePlan()` throws instead of returning `hasAccess: false` — blocked users get error page instead of blur preview | High | Implementation explicitly returns `{ hasAccess: false }` — only throws on unauthenticated (null membership). Acceptance criteria verifies `hasAccess: plans.includes(effectivePlan)` is present. |
| Null subscription treated as paid plan — org with no Subscription row gains access | High | `effectivePlan` defaults to `'starter'` when `sub === null`. Verified by `ACTIVE_STATUSES.includes(sub.status)` which is false for null (short-circuit). |
| `trialing` status with non-growth plan grants access — trial user on starter accesses growth features | High | `effectivePlan` is derived from `sub.plan`, not from `sub.status`. A trialing starter user has `plan: 'starter'` → `effectivePlan = 'starter'` → `hasAccess: false`. |
| CSS blur bypass (DevTools removes blur class) — client removes `blur-sm` to view page content | Medium | Gate is enforced server-side via `requirePlan()`. The page HTML is returned blurred from the server. Removing CSS reveals already-rendered server HTML — acceptable for read-only preview data (per RESEARCH.md §Security). |
